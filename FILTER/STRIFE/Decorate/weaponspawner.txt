//Need a silenced weapon for some missions

ACTOR Silenced_Pistol : Vanilla_Pistol
{
	+WEAPON.NOALERT
	States
	{
	Fire:
		PISG A 0 A_ReplaceSprite("vvw_models","PIAG")
		"####" A 4
		"####" "#" 0 A_JumpIf(GetCVAR("vvw_weapons")==1, "Fire.Enhanced")
		"####" "#" 0 A_PlaySound("weapons/spistol", CHAN_WEAPON)
		"####" "#" 0 A_GunFlash
		"####" B 6 A_FireBullets(5.6, 0, 1, 5, "BulletPuff")
		"####" C 4
		"####" B 5 A_ReFire
		Goto Ready
	Fire.Enhanced:
		"####" "#" 0 A_PlaySound("weapons/spistol", CHAN_WEAPON) //Enhanced sound
		"####" "#" 0 A_GunFlash
		"####" B 6 A_FireBullets(5.6, 0, 1, 5, "EnBulletPuff") //Enhanced hit puffs
		"####" C 4
		"####" B 5 A_ReFire
		TNT1 A 0 A_FireProjectile("GunBarrelSmoke",0,0,0,0,FPF_NOAUTOAIM,0)
		Goto Ready
	}
}
